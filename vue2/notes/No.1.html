methods: {
  getGrayDate () {
    // У░ЃућеТјЦтЈБУјитЈќтљјтЈ░У«Йуй«уџёУ┐йТѓ╝ТЌЦТюЪУїЃтЏ┤№╝ѕУхитДІТЌЦ~ТѕфТГбТЌЦ№╝Ѕ
    this.mournDate = {
      start: 1669824000000, // 2022-12-01
      end: 1670601600000 // 2022-12-10
    }
    this.checkGray(this.mournDate)
  },
  checkGray (mournDate) {
    var now = new Date().getTime()
    if (mournDate.start <= now && now <= mournDate.end) {
      // У┐йТѓ╝ТЌЦтЈфТюЅУ«Йуй«тюеhtmlтЁЃу┤аСИіТЅЇСИЇС╝џтй▒тЊЇfixedтњїabsoluteт«џСйЇ
      this.setGray()

        this.cancelGray()
      }, mournDate.end - now)
    } else {
      this.cancelGray()
      if (now < mournDate.start) { // УЄфтіетЈўСИ║ж╗ЉуЎй
        setTimeout(() => {
          this.setGray()
        }, mournDate.start - now)
      }
    }
  },
  setGray () {
    document.documentElement.style.filter = 'grayscale(100%)'
  },
  cancelGray () {
    document.documentElement.style.filter = 'none'
  }
}
